// Function to toggle the class between 'show' and 'none'
function toggleIframeVisibility(containerId, iconId) {
    var iframeContainer = document.getElementById(containerId);
    var buttonIcon = document.getElementById(iconId);

    // Toggle the visibility of the iframe
    if (iframeContainer.classList.contains("show")) {
        iframeContainer.classList.remove("show");
        iframeContainer.classList.add("none");
        buttonIcon.classList.replace("bi-chevron-down", "bi-chevron-up");
    } else {
        iframeContainer.classList.remove("none");
        iframeContainer.classList.add("show");
        buttonIcon.classList.replace("bi-chevron-up", "bi-chevron-down");
    }
}

// Event listeners for 'From' location
document.getElementById("pinFromLocationButton").addEventListener("click", function () {
    toggleIframeVisibility("iframeFromContainer", "pinFromLocationIcon");
});

// Event listeners for 'To' location
document.getElementById("pinToLocationButton").addEventListener("click", function () {
    toggleIframeVisibility("iframeToContainer", "pinToLocationIcon");
});
