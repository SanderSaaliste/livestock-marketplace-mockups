// Function to search messages
function searchMessages() {
    const searchTerm = document.querySelector('.message-search input').value.trim().toLowerCase();
    const messageBoxes = document.querySelectorAll('.small-message-box');
    let matchCount = 0;

    messageBoxes.forEach(box => {
        const lastMessage = box.querySelector('.last-message');
        const userName = box.querySelector('.user-name');

        // Reset highlights and visibility
        lastMessage.innerHTML = lastMessage.textContent;
        userName.innerHTML = userName.textContent;
        box.style.display = '';

        // Highlight matches or hide message boxes without matches
        if (searchTerm && !(lastMessage.textContent.toLowerCase().includes(searchTerm) || userName.textContent.toLowerCase().includes(searchTerm))) {
            box.style.display = 'none';
        }
    });

    console.log(`Search results: ${matchCount} match${matchCount !== 1 ? 'es' : ''}.`);
}

// Function to get current time in the format HH:mm AM/PM
function getCurrentTime() {
    const now = new Date();
    let hours = now.getHours();
    const minutes = now.getMinutes().toString().padStart(2, '0');
    const ampm = hours >= 12 ? 'PM' : 'AM';
    hours = hours % 12 || 12;
    return `${hours}:${minutes} ${ampm}`;
}

// Function to get current date in the format DD/MMM/YYYY
function getCurrentDate() {
    const now = new Date();
    const day = now.getDate().toString().padStart(2, "0");
    const month = now.toLocaleString('default', { month: 'short' });
    const year = now.getFullYear();
    return `${day}/${month}/${year}`;
}

// Function to attach event listeners to modal components
function attachModalListeners() {
    const budgetInput = document.getElementById('budget-input');
    const dropdown = document.getElementById('small-listing-dropbox');
    const priceValue = document.getElementById('price-value');
    const weightValue = parseInt(document.getElementById('weight-value').textContent, 10); // Get the weight value from the DOM

    // Event listener for the dropdown to change the placeholder of the budget input
    dropdown.addEventListener('change', function () {
        if (dropdown.value === 'total-amount') {
            budgetInput.placeholder = 'Total';
        } else if (dropdown.value === 'per-kg') {
            budgetInput.placeholder = 'Per/kg';
        }
    });

    // Event listener for input changes to update the price-value span
    budgetInput.addEventListener('input', function () {
        const inputValue = parseFloat(budgetInput.value); // Get the input value and convert it to a number

        // Check if "Per/kg" is selected and if the input value is a valid number
        if (dropdown.value === 'per-kg' && !isNaN(inputValue)) {
            // Multiply input value by the weight and update the price-value span
            const totalPrice = inputValue * weightValue;
            priceValue.textContent = totalPrice.toLocaleString(); // Display result with thousand separators
        } else if (dropdown.value === 'total-amount') {
            // If "Total" is selected, just update the price value with the input value
            priceValue.textContent = inputValue ? inputValue.toLocaleString() : '0';
        } else {
            priceValue.textContent = '20,000'; // Reset to default value if input is cleared or no dropdown option is selected
        }
    });
}


// Array to store each rectangle's details separately
let rectanglesData = [];

// Function to generate a unique ID for each rectangle
function generateUniqueId(prefix = "MR") {
    return `${prefix}${Math.floor(Math.random() * 100000)}`;
}

// Function to handle the submit button click with event delegation
document.addEventListener('click', function (event) {
    if (event.target && event.target.classList.contains('submit-request-btn')) {
        // Find the rectangle inside the modal
        const modalRectangle = document.querySelector('.modal-rectangle-container .rectangle');

        if (modalRectangle) {
            // Create a unique ID for this rectangle
            const uniqueId = generateUniqueId();

            // Extract price and weight values from the modal-rectangle
            const weightValue = modalRectangle.querySelector('#weight-value').textContent;
            const priceValue = modalRectangle.querySelector('#price-value').textContent;

            // Save the rectangle details in the array using the unique ID
            const rectangleDetails = {
                id: uniqueId,
                weight: weightValue,
                price: priceValue
            };

            rectanglesData.push(rectangleDetails);

            // Create a new message box for the user message
            const newMessageBox = document.createElement('div');
            newMessageBox.classList.add('second-small-chat-box');

            // Create the date line
            const lineWithDate = document.createElement('div');
            lineWithDate.classList.add('line-with-date');
            lineWithDate.innerHTML = `
                <div class="line"></div>
                <span class="date">${getCurrentDate()}</span>
                <div class="line"></div>
            `;

            // Create the user message with profile, name, and time
            newMessageBox.innerHTML = `
                <div class="second-chat-image-info-wrap">
                    <div class="second-message-info">
                        <div class="second-user-name">Me</div>
                        <div class="second-chat-message-time">${getCurrentTime()}</div>
                        <div class="second-chat-message"></div>
                    </div>
                    <div class="profile-container">
                        <img src="Rectangle 405.png" alt="Profile Image" class="profile-image">
                    </div>
                </div>
            `;

            // Deep clone the modal-rectangle to prevent overwriting
            const clonedRectangle = modalRectangle.cloneNode(true);
            clonedRectangle.id = uniqueId;  // Set the unique ID for this rectangle

            // Remove any IDs or set them to null inside the cloned rectangle
            clonedRectangle.querySelectorAll('[id]').forEach(function(element) {
                element.removeAttribute('id');
            });

            // Append the cloned rectangle to the message content
            const messageContent = newMessageBox.querySelector('.second-chat-message');
            messageContent.appendChild(clonedRectangle);

            // Append the date line and message box to the chat area
            const chatArea = document.querySelector('.chat-area');
            chatArea.appendChild(lineWithDate);  // Add the date line first
            chatArea.appendChild(newMessageBox);  // Add the message box after the date

            // Scroll to the bottom of the chat area to see the new message
            chatArea.scrollTop = chatArea.scrollHeight;

            // Log the stored rectangle data for debugging
            console.log('All rectangles:', rectanglesData);
        }

        // Close the modal after appending the message
        closeNegotiationModal();
    }
});

// Function to get current date and time
function getCurrentDate() {
    const now = new Date();
    const day = now.getDate().toString().padStart(2, "0");
    const month = now.toLocaleString('default', { month: 'short' });
    const year = now.getFullYear();
    return `${day}/${month}/${year}`;
}

function getCurrentTime() {
    const now = new Date();
    let hours = now.getHours();
    const minutes = now.getMinutes().toString().padStart(2, '0');
    const ampm = hours >= 12 ? 'PM' : 'AM';
    hours = hours % 12 || 12;
    return `${hours}:${minutes} ${ampm}`;
}
