  const verticalLines = document.querySelectorAll('.vertical-lines .line');
  const horizontalLines = document.querySelectorAll('.y-axis .horizontal-line');

  verticalLines.forEach((vLine, vIndex) => {
    horizontalLines.forEach((hLine, hIndex) => {
      const dot = document.createElement('div');
      dot.classList.add('graph-dot');

      // Calculate the position
      const vLineRect = vLine.getBoundingClientRect();
      const hLineRect = hLine.getBoundingClientRect();
      const graphContainerRect = graphContainer.getBoundingClientRect();

      const left = vLineRect.left - graphContainerRect.left - 4; // Adjust for half the dot's width
      const top = hLineRect.top - graphContainerRect.top - 4; // Adjust for half the dot's height

      dot.style.left = `${left}px`;
      dot.style.top = `${top}px`;

      graphContainer.appendChild(dot);
    });
  });
});


document.addEventListener('DOMContentLoaded', function() {
    const graphContainer = document.querySelector('.graph-content');
    const verticalLines = document.querySelectorAll('.vertical-lines .line');
    const yAxisLabels = [230, 220, 210, 200, 190, 180, 170];
    const maxPrice = Math.max(...yAxisLabels);
    const minPrice = Math.min(...yAxisLabels);

    const tooltip = document.getElementById('tooltip');
    const marketPriceContainer = document.querySelector('.market-price .price-background');
    const lowestPriceContainer = document.querySelector('.market-price .lowest-price-background');

    const svgNS = "http://www.w3.org/2000/svg"; 

    const svg = document.createElementNS(svgNS, "svg");
    svg.setAttribute('class', 'graph-line-svg');
    svg.style.position = 'absolute';
    svg.style.top = 0;
    svg.style.left = 0;
    svg.style.width = '100%';
    svg.style.height = '100%';

    const line = document.createElementNS(svgNS, "polyline");
    line.setAttribute('class', 'graph-line');
    svg.appendChild(line);

    const lowestLine = document.createElementNS(svgNS, "polyline");
    lowestLine.setAttribute('class', 'lowest-graph-line');
    svg.appendChild(lowestLine);

    graphContainer.appendChild(svg);

    const dayLabels = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday'];

    function getRandomPrice() {
        return Math.floor(Math.random() * (maxPrice - minPrice + 1)) + minPrice;
    }

    function getLowestPrice(averagePrice) {
        return averagePrice - Math.floor(Math.random() * 10 + 5); // Ensure lowest price is always smaller
    }

    function updateGraph(dayIndex, prices, lowestPrices) {
        const points = [];
        const lowestPoints = [];
        const dots = graphContainer.querySelectorAll('.graph-dot');
        dots.forEach(dot => dot.remove()); // Remove existing dots

        verticalLines.forEach((vLine, index) => {
            const vLineRect = vLine.getBoundingClientRect();
            const graphContainerRect = graphContainer.getBoundingClientRect();
            const left = vLineRect.left - graphContainerRect.left + vLine.offsetWidth / 2; // Adjusting to the center of the line
            const price = prices[index];
            const lowestPrice = lowestPrices[index];
            const top = ((maxPrice - price) / (maxPrice - minPrice)) * (graphContainer.clientHeight - 40) + 10; // Adding padding
            const lowestTop = ((maxPrice - lowestPrice) / (maxPrice - minPrice)) * (graphContainer.clientHeight - 40) + 10; // Adding padding
            points.push(`${left},${top}`);
            lowestPoints.push(`${left},${lowestTop}`);

            // Add dot element for average price
            if (index <= dayIndex) {
                const dot = document.createElement('div');
                dot.classList.add('graph-dot');
                dot.style.left = `${left - 4}px`; // Adjust for half the dot's width
                dot.style.top = `${top - 4}px`; // Adjust for half the dot's height
                dot.dataset.price = price;
                dot.dataset.day = dayLabels[index];

                dot.addEventListener('mouseover', function() {
                    tooltip.style.display = 'block';
                    tooltip.innerText = `${dot.dataset.day} ${dot.dataset.price}`;
                    tooltip.style.left = `${dot.style.left}`;
                    tooltip.style.top = `${parseInt(dot.style.top) - 20}px`; // Position above the dot
                });

                dot.addEventListener('mouseout', function() {
                    tooltip.style.display = 'none';
                });

                graphContainer.appendChild(dot);
            }

            // Add dot element for lowest price
            if (index <= dayIndex) {
                const dot = document.createElement('div');
                dot.classList.add('graph-dot');
                dot.style.left = `${left - 4}px`; // Adjust for half the dot's width
                dot.style.top = `${lowestTop - 4}px`; // Adjust for half the dot's height
                dot.dataset.price = lowestPrice;
                dot.dataset.day = dayLabels[index];

                dot.addEventListener('mouseover', function() {
                    tooltip.style.display = 'block';
                    tooltip.innerText = `${dot.dataset.day} ${dot.dataset.price}`;
                    tooltip.style.left = `${dot.style.left}`;
                    tooltip.style.top = `${parseInt(dot.style.top) - 20}px`; // Position above the dot
                });

                dot.addEventListener('mouseout', function() {
                    tooltip.style.display = 'none';
                });

                graphContainer.appendChild(dot);
            }
        });

        line.setAttribute('points', points.slice(0, dayIndex + 1).join(' '));
        lowestLine.setAttribute('points', lowestPoints.slice(0, dayIndex + 1).join(' '));

        // Update the market price
        marketPriceContainer.innerHTML = `
            <img width="20" height="20" src="https://img.icons8.com/color/96/peso-symbol.png" alt="peso-symbol"/>
            ${prices[dayIndex]}
        `;

        // Update the lowest price
        lowestPriceContainer.innerHTML = `
            <img width="20" height="20" src="https://img.icons8.com/color/96/peso-symbol.png" alt="peso-symbol"/>
            ${lowestPrices[dayIndex]}
        `;
    }

    function startUpdateCycle() {
        const prices = Array.from({ length: 7 }, getRandomPrice);
        const lowestPrices = prices.map(price => getLowestPrice(price));
        let dayIndex = 0;

        function updateDay() {
            updateGraph(dayIndex, prices, lowestPrices);
            dayIndex = (dayIndex + 1) % 7;
            if (dayIndex === 0) {
                prices.length = 0;
                lowestPrices.length = 0;
                prices.push(...Array.from({ length: 7 }, getRandomPrice));
                lowestPrices.push(...prices.map(price => getLowestPrice(price)));
            }
        }

        updateDay(); // Initial graph update
        setInterval(updateDay, 5000); // Update graph every 5 seconds
    }

    startUpdateCycle();
});

    const reviewBoxes = document.querySelectorAll('.review-box');
    const searchResult = document.querySelector('.review-search-result');

    searchButton.addEventListener('click', function() {
        const searchTerm = searchBox.value.trim().toLowerCase();
        let matchCount = 0;

        // Clear previous highlights and show all reviews
        reviewBoxes.forEach(box => {
            const description = box.querySelector('.review-description');
            const originalText = description.textContent;
            description.innerHTML = originalText; // Remove any previous highlights
            box.style.display = ''; // Show all reviews initially
        });

        if (searchTerm) {
            reviewBoxes.forEach(box => {
                const description = box.querySelector('.review-description');
                const originalText = description.textContent.toLowerCase();
                let highlightedText = description.textContent;
                const regex = new RegExp(`(${searchTerm})`, 'gi'); // Case-insensitive match

                if (originalText.includes(searchTerm)) {
                    matchCount += (originalText.match(regex) || []).length; // Count matches
                    highlightedText = description.textContent.replace(regex, '<span class="highlight">$1</span>'); // Highlight matches
                    description.innerHTML = highlightedText; // Update description with highlighted text
                } else {
                    box.style.display = 'none'; // Hide reviews without the search term
                }
            });

            // Display the search result with correct grammar
            const resultText = `Search results: The word '${searchTerm}' was found ${matchCount} time${matchCount !== 1 ? 's' : ''}.`;
            searchResult.textContent = resultText;
        } else {
            searchResult.textContent = ''; // Clear result if search term is empty
            reviewBoxes.forEach(box => box.style.display = ''); // Show all reviews if search term is empty
        }
    });
});
